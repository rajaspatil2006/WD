    <script>
        function validateForm() {

            var name = document.forms["myForm"]["name"].value;
            var email = document.forms["myForm"]["email"].value;

            if (name == "") {
                alert("Name must be filled out");
                rturn false;
            }
            if (email == "") {
                alert("Email must be filled out");
                return false;
            }
            if (!email.includes("@")) {
                alert("Invalid Email");
                return false;
            }
            alert("Form Submitted Successfully");
            return true;
        }
    </script>
